cmake_minimum_required(VERSION 3.10)
project(Inversecudamatics)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Znajdź yaml-cpp
find_package(yaml-cpp REQUIRED)
find_package(Eigen3 REQUIRED)

# Dodaj podprojekty
add_subdirectory(kinematicbody)
# Tworzymy program główny
add_executable(Inversecudamatics main.cpp)

# Linkowanie bibliotek
target_link_libraries(Inversecudamatics
    PRIVATE
        yaml-cpp
        Eigen3::Eigen
        robotmodel
        kinematic
        yamlloader
        
        
)
